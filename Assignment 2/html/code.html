%%task 1
v = rand(2, 1);


unitV = v / norm(v);

M = zeros(2,6);
M(:,1) = unitV;

for i=2:6
    p = (i-1)*(pi/3);
    R = [cos(p) -sin(p); sin(p) cos(p)];
    M(:,i) = R*unitV ;
end

Display

figure(1)
origin = zeros(1,size(M,2));
plot([origin; M(1,:)],[origin; M(2,:)]);
