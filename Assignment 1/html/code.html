function [list, ld, li] = getCase(row,col)
    myList = {};
    ld = 0;
    li = 0;
    for i=1:5
        vectors = rand(row,col);
        theRank = rank(vectors);
        disp(theRank)
        myList = [myList, vectors];
        if (theRank == row)
            li = li + 1;
        else
            ld = ld + 1;
        end

    end
    list = myList;
end
